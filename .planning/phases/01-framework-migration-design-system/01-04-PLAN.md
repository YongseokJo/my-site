   ```astro
   import BaseLayout from "../layouts/BaseLayout.astro";

   const allPages = await getCollection("pages");
   const aboutPage = allPages.find(p => p.id === "about" || p.id === "about.md");

   if (aboutPage) {
     const rendered = await aboutPage.render();
     Content = rendered.Content;
   }

     <h1 class="text-3xl font-bold mb-8">About</h1>
     {Content ? (
       <div class="prose prose-invert max-w-none"><Content /></div>
     ) : (
       <p class="text-muted-foreground">Content coming soon.</p>
     )}
   </BaseLayout>
   ```
   Uses `aboutPage.render()` to get proper HTML from Markdown content. The `body` property returns raw Markdown text — `render()` is required for glob()-loaded Content Collections in Astro 5.x.
