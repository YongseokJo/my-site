Create `src/pages/arxiv-app/tutorials/[slug].astro` — the dynamic route for tutorial pages per D-14 and UI-SPEC "Readea Tutorials" layout.

This page uses Astro's `getStaticPaths()` with the tutorials content collection:

```astro
---
import BaseLayout from "../../../layouts/BaseLayout.astro";
import Breadcrumb from "../../../components/Breadcrumb.astro";
import TutorialNav from "../../../components/TutorialNav.astro";
import { getCollection } from "astro:content";

export async function getStaticPaths() {
  const tutorials = await getCollection("tutorials");
  const sorted = tutorials.sort((a, b) => a.data.order - b.data.order);
  return sorted.map((tutorial, index) => ({
    params: { slug: tutorial.id },
    props: {
      tutorial,
      prev: index > 0
        ? { title: sorted[index - 1].data.title, href: `/arxiv-app/tutorials/${sorted[index - 1].id}` }
        : undefined,
      next: index < sorted.length - 1
        ? { title: sorted[index + 1].data.title, href: `/arxiv-app/tutorials/${sorted[index + 1].id}` }
        : undefined,
    },
  }));
}

const { tutorial, prev, next } = Astro.props;
const { Content } = await tutorial.render();
---

<BaseLayout title={tutorial.data.title} description={tutorial.data.description}>
  <Breadcrumb items={[
    { label: "Readea", href: "/arxiv-app" },
    { label: "Tutorials", href: "/arxiv-app" },
    { label: tutorial.data.title },
  ]} />

  <h1 class="text-2xl font-bold mt-6 mb-8">{tutorial.data.title}</h1>

  <div class="prose-like space-y-6 text-base text-foreground [&_h2]:text-xl [&_h2]:font-bold [&_h2]:mt-8 [&_h2]:mb-4 [&_p]:leading-relaxed">
    <Content />
  </div>

  <TutorialNav prev={prev} next={next} />
</BaseLayout>
```

Key implementation details:
- `getStaticPaths()` sorts tutorials by `order` field and computes prev/next for each
- Breadcrumb shows "Readea > Tutorials > {title}" per UI-SPEC
- Tutorial content is rendered via `<Content />` from the markdown file
- Markdown headings (h2 for each step) are styled via descendant selectors on the prose container
- TutorialNav at bottom shows prev/next links
- Each tutorial is a separate page per D-14
