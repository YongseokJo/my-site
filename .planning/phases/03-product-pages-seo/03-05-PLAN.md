**1. Create `src/components/SchemaOrgPublication.astro`** per RESEARCH Pattern 4 and UI-SPEC SchemaOrgPublication contract:

```astro
---
interface Props {
  title: string;
  authors: string;
  year: number;
  venue: string;
  links: { arxiv?: string; doi?: string; ads?: string; pdf?: string };
}
const { title, authors, year, venue, links } = Astro.props;

// Parse semicolon-separated author string into Person objects
const authorList = authors
  .split(";")
  .map((a) => a.trim())
  .filter((a) => a.length > 0)
  .map((name) => ({
    "@type": "Person" as const,
    name,
  }));

// Build JSON-LD — omit properties when data is missing (per UI-SPEC contract #4)
const jsonLd: Record<string, unknown> = {
  "@context": "https://schema.org",
  "@type": "ScholarlyArticle",
  name: title,
  author: authorList,
  datePublished: String(year),
};

// Add URL: prefer DOI, fallback to arXiv (per RESEARCH Pattern 4)
if (links.doi && links.doi.length > 0) {
  jsonLd.url = links.doi;
} else if (links.arxiv && links.arxiv.length > 0) {
  jsonLd.url = links.arxiv;
}

// Add publisher only if venue exists and is non-empty
if (venue && venue.length > 0) {
  jsonLd.publisher = { "@type": "Organization", name: venue };
}
---
<script type="application/ld+json" set:html={JSON.stringify(jsonLd)} />
```

Key details per UI-SPEC:
- Never output `"url": ""` or `"publisher": null` — omit entirely if missing
- Authors parsed from semicolon-separated string (matches publications.yaml format)
- `@type: "ScholarlyArticle"` per D-15 and D-17 (standard, no special Scholar optimization)

**1. Create `src/islands/SearchDialog.tsx`** — Custom Pagefind search modal per RESEARCH Pattern 2 and UI-SPEC Pagefind Search interaction contract:

```typescript
import { useState, useEffect, useRef, useCallback } from "react";
import {
  Dialog,
  DialogContent,
  DialogHeader,
  DialogTitle,
} from "@/components/ui/dialog";
import { Input } from "@/components/ui/input";
```

Component: `SearchDialog({ open, onOpenChange }: { open: boolean; onOpenChange: (open: boolean) => void })`

State:
- `query: string` — search input
- `results: SearchResult[]` — array of `{ url: string; title: string; excerpt: string }`
- `pagefindRef: useRef<any>(null)` — lazy-loaded Pagefind instance

Pagefind initialization (per RESEARCH — lazy load only when dialog opens):
```typescript
async function initPagefind() {
  if (!pagefindRef.current) {
    try {
      // @ts-ignore — Pagefind JS is generated at build time
      pagefindRef.current = await import("/pagefind/pagefind.js");
      pagefindRef.current.init();
    } catch {
      console.warn("Pagefind not available — run npm run build first");
    }
  }
}
```

Search function (debounced by Pagefind internally):
```typescript
async function handleSearch(query: string) {
  setQuery(query);
  if (!query.trim()) { setResults([]); return; }
  await initPagefind();
  if (!pagefindRef.current) return;
  const search = await pagefindRef.current.debouncedSearch(query, {}, 300);
  if (!search) return;
  const items = await Promise.all(
    search.results.slice(0, 10).map(async (r: any) => {
      const data = await r.data();
      return {
        url: data.url,
        title: data.meta?.title ?? "",
        excerpt: data.excerpt ?? "",
      };
    })
  );
  setResults(items);
}
```

JSX (per UI-SPEC layout):
```tsx
<Dialog open={open} onOpenChange={onOpenChange}>
  <DialogContent className="max-w-[600px] max-h-[70vh] flex flex-col p-0">
    <DialogHeader className="px-4 pt-4 pb-2">
      <DialogTitle className="sr-only">Search site</DialogTitle>
      <Input
        placeholder="Search..."
        value={query}
        onChange={(e) => handleSearch(e.target.value)}
        autoFocus
        className="text-base"
      />
    </DialogHeader>
    <div className="flex-1 overflow-y-auto px-4 pb-4">
      {query && results.length === 0 && (
        <p className="text-sm text-muted-foreground py-4 text-center">
          No results found for '{query}'
        </p>
      )}
      {results.map((result) => (
        <a
          key={result.url}
          href={result.url}
          onClick={() => onOpenChange(false)}
          className="block py-3 border-b border-border last:border-0 hover:bg-muted/50 -mx-4 px-4 transition-colors"
        >
          <p className="text-base font-bold text-foreground">{result.title}</p>
          <p
            className="text-sm text-muted-foreground mt-1 line-clamp-2"
            dangerouslySetInnerHTML={{ __html: result.excerpt }}
          />
        </a>
      ))}
    </div>
  </DialogContent>
</Dialog>
```

Note: `dangerouslySetInnerHTML` is used ONLY for Pagefind excerpts which contain `<mark>` tags for search term highlighting. This is safe because Pagefind sanitizes its own output (per RESEARCH security domain).

**2. Create `src/islands/SearchToggle.tsx`** — Search icon button with Cmd+K shortcut:

```typescript
import { useState, useEffect } from "react";
import { Search } from "lucide-react";
import SearchDialog from "./SearchDialog";

export default function SearchToggle() {
  const [open, setOpen] = useState(false);

  useEffect(() => {
    function handleKeyDown(e: KeyboardEvent) {
      if ((e.metaKey || e.ctrlKey) && e.key === "k") {
        e.preventDefault();
        setOpen(true);
      }
    }
    document.addEventListener("keydown", handleKeyDown);
    return () => document.removeEventListener("keydown", handleKeyDown);
  }, []);

  return (
    <>
      <button
        onClick={() => setOpen(true)}
        className="flex items-center justify-center h-10 w-10 rounded-md text-muted-foreground hover:text-foreground hover:bg-muted transition-colors duration-150"
        aria-label="Search site"
      >
        <Search className="size-5" />
      </button>
      <SearchDialog open={open} onOpenChange={setOpen} />
    </>
  );
}
```

Per UI-SPEC: search icon placed in nav bar, after nav links, before dark mode toggle. Keyboard shortcut Cmd+K / Ctrl+K. `aria-label="Search site"`.
